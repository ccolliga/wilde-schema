    <p> [http://www.w3.org/1999/xhtml]

    <p> General container tag for paragraphs, headings, and signatures.
    Namespacehttp://www.w3.org/1999/xhtml
    Modulexhtml5
    Attributesatt.id (@id)
    class
    Status Optional
    Legal values are:
    heading
    signature
    Contained by
    xhtml5: div
    May contain
    a
    xhtml5: b br i
    character data
    Schematron

    <sch:ns uri="http://www.w3.org/1999/xhtml"
     prefix="xh"/>
    <sch:let name="text"
     value="normalize-space(.)"/>
    <sch:let name="last"
     value="substring($text, string-length($text))"/>
    <sch:assert test="@class or matches($last, '^[.!?]$') or ($text = '[...]')"> Paragraphs should not end in "<sch:value-of select="$last"/>" instead of
    . ! or ?. </sch:assert>
    Schematron

    <sch:ns uri="http://www.w3.org/1999/xhtml"
     prefix="xh"/>
    <sch:let name="text"
     value="normalize-space(string())"/>
    <sch:assert test="not(matches($text, '\[\s*\.\.\.\s*[^\]]'))"> Open bracket
    [ must have closing braket ] </sch:assert>
    <sch:assert test="not(matches($text, '[^\[]\s*\.\.\.\s*\]'))"> Close bracket
    ] does not correspond to open bracket [ </sch:assert>
    <sch:assert test="not(matches($text, '(\[\s+\.\.\.)|(\.\.\.\s+\])'))"> Spaces are not allowed between brackets and elipsis </sch:assert>
    <sch:assert test="not(matches($text, '(\.\s+\.\.)|(\.\.\s+\.)'))"> Spaces
    are not allowed inside elipsis </sch:assert>
    <sch:assert test="not($text = '[...]')"> Omitted/unrelated paragraphs should
    not be indicated with <p>[...]</p>
    </sch:assert>
    Content model
    <content>
     <sequence>
      <alternate minOccurs="0"
       maxOccurs="unbounded">
       <textNode/>
       <classRef key="model.phrase"/>
      </alternate>
      <elementRef key="a" minOccurs="0"
       maxOccurs="unbounded"/>
     </sequence>
    </content>
    Schema Declaration
    element p
{

   attribute class { "heading" | "signature" }?,
